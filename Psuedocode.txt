Constants
----------------------------
DEFINE wordsBank = ["HTML", "CSS", "JAVASCRIPT", "JAVA", "PYTHON"]
DEFINE maxAttempts = 4
DEFINE hangmanImages = ["images/image1.png", "images/image2.png", "images/image3.png", "images/image4.png"]

------------------------------
Variables (state)
SET guessedWord = []
SET attempts = 0
SET imageIndex = 0
SET wordToGuess = ""

-------------------------------
Cached Element References

GET displayWord ELEMENT BY ID 'word-board'
GET messageEl ELEMENT BY ID 'message'
GET letterBtns ELEMENTS BY CLASS 'letter'
GET imageEl ELEMENT BY TAG 'img'
GET attemptsEl ELEMENT BY ID 'attempts'
--------------------------------------

Functions
------------------------------------
FUNCTION init
    SET randomWordIndex = RANDOM_INDEX_FROM(wordsBank)
    SET wordToGuess = wordsBank[randomWordIndex]
    SET guessedWord = ARRAY_OF_UNDERSCORES_WITH_LENGTH(wordToGuess.length)
    SET attempts = 0
    SET imageIndex = 0
    CALL updateDisplayWord
    CALL updateHangmanImage
    CALL updateAttempts
    ENABLE all letter buttons
    CALL updateMessage WITH ""
END FUNCTION


updateDisplayWord

FUNCTION updateDisplayWord
    SET displayWord.textContent = JOIN(guessedWord, ' ')
END FUNCTION



updateMessage

FUNCTION updateMessage WITH message
    SET messageEl.textContent = message
END FUNCTION





updateAttempts

FUNCTION updateAttempts
    SET attemptsEl.textContent = "Attempts: " + (maxAttempts - attempts)
END FUNCTION


disableBtn

FUNCTION disableBtn
    DISABLE all letter buttons
END FUNCTION




handleLetterClick

FUNCTION handleLetterClick WITH event
    SET letter = event.target.textContent
    SET isCorrectGuess = FALSE
    FOR EACH character IN wordToGuess
        IF character EQUALS letter
            SET guessedWord[INDEX] = letter
            SET isCorrectGuess = TRUE
    CALL updateDisplayWord
    IF isCorrectGuess
        IF guessedWord DOES NOT CONTAIN '_'
            CALL updateMessage WITH "You won!"
            CALL disableBtn
    ELSE
        INCREMENT attempts
        INCREMENT imageIndex
        CALL updateHangmanImage
        CALL updateAttempts
        IF attempts GREATER THAN OR EQUAL TO maxAttempts
            CALL updateMessage WITH "Game over! The word was " + wordToGuess
            CALL disableBtn
        ELSE
            CALL updateMessage WITH "Try again!"
    DISABLE event.target
END FUNCTION



handlePlayAgainClick

FUNCTION handlePlayAgainClick
    CALL init
END FUNCTION

---------------------------------------
Event Listeners

FOR EACH button IN letterBtns
    ADD EVENT LISTENER 'click' TO button WITH handleLetterClick
-*-------------------------
CALL init
ADD EVENT LISTENER 'click' TO playAgainButton WITH handlePlayAgainClick
